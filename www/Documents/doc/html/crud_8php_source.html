383function cropImage($image_path, $thumb_width, $thumb_height)
384{
385 $username = basename(dirname($image_path));
386 $image = createImageResource($image_path);
387 if ($image === false) {
388 return false;
389 }
390 $image_width = imagesx($image);
391 $image_height = imagesy($image);
392 $original_aspect = $image_width / $image_height;
393 $thumb_aspect = $thumb_width / $thumb_height;
394 if ($original_aspect >= $thumb_aspect) {
395 $new_height = $thumb_height;
396 $new_width = $image_width / ($image_height / $thumb_height);
397 } else {
398 $new_width = $thumb_width;
399 $new_height = $image_height / ($image_width / $thumb_width);
400 }
401 $thumb = imagecreatetruecolor($thumb_width, $thumb_height);
402 imagecopyresampled(
403 $thumb,
404 $image,
405 0 - ($new_width - $thumb_width) / 2,
406 0 - ($new_height - $thumb_height) / 2,
407 0,
408 0,
409 $new_width,
410 $new_height,
411 $image_width,
412 $image_height
413 );
414
415 $originalName = pathinfo($image_path, PATHINFO_FILENAME);
416 $extension = pathinfo($image_path, PATHINFO_EXTENSION);
